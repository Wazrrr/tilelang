extern "C" __global__ void tutorial_vec_add_kernel(float* __restrict__ out, const float* __restrict__ x, const float* __restrict__ y);
extern "C" __global__ void tutorial_vec_sub_kernel(float* __restrict__ out, const float* __restrict__ x, const float* __restrict__ y);
extern "C" __global__ void __launch_bounds__(128, 1) tutorial_vec_add_kernel(float* __restrict__ out, const float* __restrict__ x, const float* __restrict__ y) {
  out[((int)blockIdx.x)] = (x[((int)blockIdx.x)] + y[((int)blockIdx.x)]);
}

extern "C" __global__ void __launch_bounds__(128, 1) tutorial_vec_sub_kernel(float* __restrict__ out, const float* __restrict__ x, const float* __restrict__ y) {
  out[((int)blockIdx.x)] = (x[((int)blockIdx.x)] - y[((int)blockIdx.x)]);
}
